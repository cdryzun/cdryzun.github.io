    cfdisk /dev/vdb     # 将磁盘创建分区  为8e类型（ LVM 类型）,如果虚拟机找不到所添加的磁盘，请尝试重启解决。

# 下列样式中 创建了 /dev/vdb1 分区
sudo udevadm trigger
cat /proc/partitions
sudo pvcreate /dev/vdb1
sudo pvdisplay | grep "VG Name"   # 查看 vg 名字

vgextend centos /dev/vdb1   # 将分区纳入 vg

lvextend -l +100%FREE /dev/mapper/centos-root   # 全部加入
lvextend -L+20G /dev/mapper/centos-root              # 按需添加，这里是加入 20G

xfs_growfs /

# 使其生效 上面这条是 xfs 的， 其他文件格式尝试 resize2fs
